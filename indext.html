function applyColors(c) {

  document.documentElement.style.setProperty('--c-bg', c.background_color||defaultConfig.background_color);

  document.documentElement.style.setProperty('--c-surface', c.surface_color||defaultConfig.surface_color);

  document.documentElement.style.setProperty('--c-text', c.text_color||defaultConfig.text_color);

  document.documentElement.style.setProperty('--c-primary', c.primary_color||defaultConfig.primary_color);

  document.documentElement.style.setProperty('--c-secondary', c.secondary_color||defaultConfig.secondary_color);

  document.body.style.background = c.background_color||defaultConfig.background_color;

  document.body.style.color = c.text_color||defaultConfig.text_color;

}


function applyFonts(c) {

  const ff = c.font_family||defaultConfig.font_family;

  const fs = c.font_size||defaultConfig.font_size;

  document.body.style.fontFamily = `${ff}, DM Sans, sans-serif`;

  document.body.style.fontSize = fs+'px';

}

// ===== DATA SDK =====

const dataHandler = {

  onDataChanged(data) {

    const rec = data.find(d=>d.type==='progress');

    if(rec) {

      progressData = rec;

      localScores.vocab_score = rec.vocab_score||0;

      localScores.grammar_score = rec.grammar_score||0;

      localScores.pronunciation_done = rec.pronunciation_done||0;

      localScores.practice_count = rec.practice_count||0;

    }

    if(currentTab===5) renderProgress();

  }

};

(async()=>{

  const r = await window.dataSdk.init(dataHandler);

  if(!r.isOk) console.error('Data SDK init failed');

})();


async function saveProgress() {

  const obj = {

    type:'progress',

    vocab_score: localScores.vocab_score,

    grammar_score: localScores.grammar_score,

    pronunciation_done: localScores.pronunciation_done,

    practice_count: localScores.practice_count,

    last_updated: new Date().toISOString()

  };

  if(progressData) {

    const updated = {...progressData,...obj};

    const r = await window.dataSdk.update(updated);

    if(!r.isOk) console.error('Update failed');

  } else {

    const r = await window.dataSdk.create(obj);

    if(!r.isOk) console.error('Create failed');

  }

}


// ===== LANGUAGE =====

function toggleLang(){

  lang = lang==='en'?'vn':'en';

  document.getElementById('langLabel').textContent = lang==='en'?'VN':'EN';

  document.querySelectorAll('[data-en]').forEach(el=>{

    el.textContent = el.getAttribute('data-'+lang);

  });

  renderCurrentSection();

}

function t(en,vn){ return lang==='en'?en:vn; }


// ===== TABS =====

function switchTab(i){

  document.querySelectorAll('#tabBar .tab-btn').forEach((b,idx)=>b.classList.toggle('active',idx===i));

  for(let s=0;s<6;s++){

    document.getElementById('sec'+s).classList.toggle('section-hidden',s!==i);

  }

  currentTab=i;

  renderCurrentSection();

}

function renderCurrentSection(){

  [renderQuestions,renderVocab,renderGrammar,renderPronunciation,renderPractice,renderProgress][currentTab]();

}


// ===== TTS =====

function speak(text){

  if('speechSynthesis' in window){

    window.speechSynthesis.cancel();

    const u=new SpeechSynthesisUtterance(text);

    u.lang='en-US'; u.rate=0.85;

    window.speechSynthesis.speak(u);

  }

}


// ===== TOAST =====

function showToast(msg,color='#22c55e'){

  const d=document.createElement('div');

  d.className='toast';d.style.background=color;d.textContent=msg;

  document.body.appendChild(d);

  setTimeout(()=>d.remove(),2500);

}

function renderVocabQuiz(){

  const allWords = vocabData.flatMap(c=>c.words);

  const selected = allWords.sort(()=>Math.random()-.5).slice(0,4);

  const meanings = selected.map(w=>w.meaning).sort(()=>Math.random()-.5);

  vocabQuizState = {selected, answers:{}, meanings};


  const container = document.getElementById('vocabQuiz');

  if(!container)return;

  container.innerHTML = selected.map((w,i)=>`

    <div class="card" id="vq${i}">

      <p class="font-bold mb-2">${w.word}</p>

      <div class="space-y-2">

        ${meanings.map((m,mi)=>`

          <div class="quiz-option" onclick="checkVocabQuiz(${i},${mi},'${m.replace(/'/g,"\\'")}')">${m}</div>

        `).join('')}

      </div>

    </div>

  `).join('');

}


function checkVocabQuiz(qi,mi,chosen){

  const w = vocabQuizState.selected[qi];

  const card = document.getElementById('vq'+qi);

  const opts = card.querySelectorAll('.quiz-option');

  opts.forEach((o,idx)=>{

    if(o.textContent===w.meaning) o.classList.add('correct');

    else if(idx===mi && chosen!==w.meaning) o.classList.add('wrong');

    o.style.pointerEvents='none';

  });

  if(chosen===w.meaning){

    localScores.vocab_score = Math.min(100, localScores.vocab_score+10);

    showToast(t('✅ Correct!','✅ Đúng rồi!'));

    saveProgress();

  } else {

    showToast(t('❌ Try again next time!','❌ Lần sau nhé!'),'#ef4444');

  }

}

function analyzePractice(i){

  const textarea = document.getElementById('pAns'+i);

  const answer = textarea.value.trim();

  practiceAnswers[i] = answer;

  if(!answer){

    showToast(t('Please write an answer first','Hãy viết câu trả lời trước'),'#f59e0b');

    return;

  }


  const words = answer.split(/\s+/).length;

  const sentences = answer.split(/[.!?]+/).filter(s=>s.trim()).length;


  // Simple analysis

  const discourseMarkers = ['in my opinion','i think','on the other hand','for example','however','moreover','in addition','as a result','therefore','firstly','secondly','in my view','personally'];

  const usedMarkers = discourseMarkers.filter(m=>answer.toLowerCase().includes(m));


  const vocabWords = vocabData.flatMap(c=>c.words.map(w=>w.word.toLowerCase()));

  const usedVocab = vocabWords.filter(v=>answer.toLowerCase().includes(v));


  const grammarPatterns = ['have been','has been','have changed','used to','will be','going to','would be','if i had','because','as a result'];

  const usedGrammar = grammarPatterns.filter(p=>answer.toLowerCase().includes(p));


  let vocabLevel = words<20?t('⚠️ Basic - try using more vocabulary','⚠️ Cơ bản - thử dùng thêm từ vựng'):

    usedVocab.length>0?t('✅ Good - you used topic vocabulary!','✅ Tốt - bạn đã dùng từ vựng chủ đề!'):

    t('💡 Try including words from the Vocabulary section','💡 Thử dùng từ từ phần Từ vựng');


  let grammarLevel = usedGrammar.length>0?t('✅ Good grammar structures detected','✅ Phát hiện cấu trúc ngữ pháp tốt'):

    t('💡 Try using structures from the Grammar section','💡 Thử dùng cấu trúc từ phần Ngữ pháp');


  let ideaLevel = sentences>=2?t('✅ Good idea development','✅ Phát triển ý tốt'):

    t('💡 Try to give reasons and examples','💡 Thử đưa ra lý do và ví dụ');


  const fb = document.getElementById('pFeedback'+i);

  fb.innerHTML = `

    <div class="p-4 rounded-xl text-sm" style="background:var(--c-secondary)">

      <p class="font-bold mb-2">${t('📝 Feedback','📝 Phản hồi')}</p>

      <div class="space-y-2">

        <p><b>${t('Words','Số từ')}:</b> ${words} ${words<30?t('(Try 40+ words)','(Thử 40+ từ)'):t('(Good length!)','(Độ dài tốt!)')}</p>

        <p><b>${t('Vocabulary','Từ vựng')}:</b> ${vocabLevel}</p>

        <p><b>${t('Grammar','Ngữ pháp')}:</b> ${grammarLevel}</p>

        <p><b>${t('Ideas','Ý tưởng')}:</b> ${ideaLevel}</p>
